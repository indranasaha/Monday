pipeline {
    agent any 
    stages {
        stage('clone repo and clean it') { 
            steps {
               bat "mvn clean -f Monday"
            }
        }
        stage('Test') { 
            steps {
                bat "mvn test -f Monday"
            }
        }
        stage('Deploy') { 
            steps {
                bat "mvn package -f Monday"
            }
        }
    }
}
